Cause : "playerShip" is searched even though it's called "PlayerShip"
GameObject* PlayField::GetPlayerObject()
{
	auto it = std::find_if(gameObjects.begin(), gameObjects.end(), [](GameObject* in) { return (strcmp(in->m_objType, "playerShip") == 0); });
	if (it != gameObjects.end())
		return (*it);
	else
		return nullptr;
}
Solution : replace "playerShip" with "PlayerShip"

Problème : crash on PlayerLaser destructor
PlayerLaser::~PlayerLaser()
{
	delete[] m_objType;
}
Cause : PlayerLaser is destructed in its Update loop, causing the gameObjects list to change directly during iteration
void PlayerLaser::Update(PlayField& world)
{
	bool deleted = false;
	pos.y -= 1.f;
	if (pos.y < 0)
	{
		deleted = true;
	}

	if (deleted)
	{
		world.DespawnLaser(this);
		delete this;
	}
}
Solution : Remove "delete this" as "world.DespawnLaser(this);" already takes care of cleanly despawning the lazer.

Problème : aliens spawn in a boring straight line

	for (int k = 0; k < 20; k++)

Problème : hit Aliens do not disappear
Cause : nothing checks for their death
Solution : add this to PlayerLaser Update
auto currentGameObjects = world.GameObjects();
for (auto& it : currentGameObjects)
{
	if (pos.IntCmp(it->pos) && strcmp(it->m_objType, "AlienShip") == 0)
	{
		deleted = true;
		auto shouldDie = it->DecreaseHealth();
		if (shouldDie)
			world.RemoveObject(it);
	}
}
